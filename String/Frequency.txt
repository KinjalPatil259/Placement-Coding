#include<stdio.h>
int main()
{

    char str[100];
    int count=0;
    char ch;

    printf("enter string :- ");
    scanf("%s",str);

    printf("enter character :-");
    scanf(" %c",&ch);

    for(int i=0;str[i]!='\0';i++)
    {
        if(str[i]==ch)
        {
            count++;
        }
    }
    printf("count : - %d",count);
    return 0;
}
