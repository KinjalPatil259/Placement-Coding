#include<stdio.h>
int main()
{
    int n,a[10],largest;

    printf("enter size of the array = ");
    scanf("%d",&n);

    for(int i=0;i<n;i++)
    {
        printf("a[%d]=",i);
        scanf("%d",&a[i]);
    }

    largest=a[0];

    for(int i=1;i<n;i++)
    {
        if(a[i]>largest)
        {
            largest=a[i];
        }
        
    }
    printf("largest= %d",largest);
 
    return 0;
}
