#include<stdio.h>
int main()
{
    int n1,n2,gcd,lcm;

    printf("enter number1=");
    scanf("%d",&n1);
    printf("enter number2=");
    scanf("%d",&n2);

    for(int i=1;i<=n1 && i<=n2;i++)
    {
        if(n1%i==0 && n2%i==0)
        {
            gcd=i;
        }
    }

    printf("gcd=%d",gcd);

    lcm=(n1*n2)/gcd;
    printf("\nlcm=%d",lcm);
    return 0;
}
