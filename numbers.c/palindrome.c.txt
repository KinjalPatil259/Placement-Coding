#include<stdio.h>
int main()
{
    int n,rem,reverse=0,temp;
    printf("enter number=");
    scanf("%d",&n);
    temp=n;
    while(n!=0)
    {
        rem=n%10;
        reverse=reverse*10+rem;
        n=n/10;
    }
    if(temp==reverse)
    {
        printf("%d is palindrome",temp);
    }
    else
    {
         printf("%d is not  palindrome",temp);
       
    }
    return 0;
}
